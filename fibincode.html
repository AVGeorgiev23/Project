Fibonacci sequence in C++

    #include 
    using namespace std;
    int main()
    {
        int a = 0;
        int b = 1;
        unsigned int sum = 0;
    
        for (int i = 0; i <= 100; i++) {
            sum = a + b;
            cout << i << ": " << sum << endl;
            a = b;
            b = sum;
        }
    
    }
